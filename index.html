// Autenticação async function checkAuth() { const token = localStorage.getItem('authToken'); if (!token) return; try { const response = await callApi({ path: '/auth/validate', method: 'POST', data: { token } }); if (!response.valid) { localStorage.removeItem('authToken'); return; } currentUser = { email: response.user.email, nome: response.user.nome, isProati: response.user.perfil === 'PROATI', token }; initApp(); } catch (error) { console.error('Erro ao validar token:', error); localStorage.removeItem('authToken'); } } async function handleLogin() { const email = emailInput.value.trim(); const password = passwordInput.value.trim(); // Reset erros emailError.classList.add('hidden'); passwordError.classList.add('hidden'); if (!email) { showError(emailError, 'Por favor, insira seu e-mail'); return; } if (!isValidEmail(email)) { showError(emailError, 'Por favor, insira um e-mail institucional válido'); return; } if (!password) { showError(passwordError, 'Por favor, insira sua senha'); return; } loginText.classList.add('hidden'); loginSpinner.classList.remove('hidden'); loginBtn.disabled = true; try { const response = await callApi({ path: '/auth/login', method: 'POST', data: { email, password } }); if (!response.token) { throw new Error(response.error || 'Falha no login'); } currentUser = { email: response.user.email, nome: response.user.nome, isProati: response.user.perfil === 'PROATI', token: response.token }; localStorage.setItem('authToken', response.token); initApp(); } catch (error) { showError(passwordError, error.message || 'Erro inesperado'); passwordInput.value = ''; } finally { loginText.classList.remove('hidden'); loginSpinner.classList.add('hidden'); loginBtn.disabled = false; } }
